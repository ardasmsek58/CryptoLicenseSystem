143std::string HardwareLock::getCpuId() {
144 std::string cpuId;
145
146#ifdef _WIN32
147 int cpuInfo[4] = {0};
148 __cpuid(cpuInfo, 0);
149 char vendor[13] = {0};
150 *reinterpret_cast<int*>(vendor) = cpuInfo[1];
151 *reinterpret_cast<int*>(vendor + 4) = cpuInfo[3];
152 *reinterpret_cast<int*>(vendor + 8) = cpuInfo[2];
153 __cpuid(cpuInfo, 1);
154 std::ostringstream oss;
155 oss << vendor << "_" << std::hex << cpuInfo[0];
156 cpuId = oss.str();
157
158#elif defined(__linux__)
159 std::ifstream file("/proc/cpuinfo");
160 if (file.is_open()) {
161 std::string line;
162 while (std::getline(file, line)) {
163 if (line.find("processor") == 0) {
164 size_t pos = line.find(":");
165 if (pos != std::string::npos) {
166 cpuId = line.substr(pos + 1);
167 break;
168 }
169 }
170 }
171 file.close();
172 }
173 if (cpuId.empty()) {
174 cpuId = executeCommand("cat /proc/cpuinfo | grep 'model name' | head -1 | cut -d':' -f2");
175 }
176
177#elif defined(__APPLE__)
178 cpuId = executeCommand("sysctl -n machdep.cpu.brand_string");
179 if (cpuId.empty()) {
180 cpuId = executeCommand("system_profiler SPHardwareDataType | grep 'Processor Name' | cut -d':' -f2");
181 }
182#endif
183
184 cpuId.erase(std::remove_if(cpuId.begin(), cpuId.end(),
185 [](char c) { return std::isspace(c) || c == '\r' || c == '\n'; }), cpuId.end());

224bool HardwareLock::verifyLicense(const std::string &hash, const std::string &signatureBase64, const std::string &publicKeyPath) {
225 qDebug() << "=== Signature Verification Started ===";
226 qDebug() << "Hash length:" << hash.length();
227 qDebug() << "Signature length:" << signatureBase64.length();
228 qDebug() << "Public key file:" << QString::fromStdString(publicKeyPath);
229
230 bool isHex = true;
231 for (char c : signatureBase64) {
232 if (!((c >= '0' && c <= '9') || (c >= 'a' && c <= 'f') || (c >= 'A' && c <= 'F'))) {
233 isHex = false;
234 break;
235 }
236 }
237
238 qDebug() << "Signature format:" << (isHex ? "HEX" : "BASE64");
239
240 std::vector<unsigned char> signature;
241 if (isHex && signatureBase64.length() == 512) {
242 signature.resize(256);
243 for (size_t i = 0; i < signatureBase64.length(); i += 2) {
244 std::string byteString = signatureBase64.substr(i, 2);
245 signature[i / 2] = (unsigned char)strtol(byteString.c_str(), nullptr, 16);
246 }
247 } else {
248 BIO *bio = nullptr;
249 BIO *b64 = nullptr;
250 try {
251 b64 = BIO_new(BIO_f_base64());
252 if (!b64) return false;
253
254 bio = BIO_new_mem_buf(signatureBase64.c_str(), signatureBase64.length());
255 if (!bio) {
256 BIO_free(b64);
257 return false;
258 }
259 bio = BIO_push(b64, bio);
260 BIO_set_flags(bio, BIO_FLAGS_BASE64_NO_NL);
261
262 std::vector<unsigned char> tempSignature(signatureBase64.length());
263 int actualLen = BIO_read(bio, tempSignature.data(), tempSignature.size());
264 if (actualLen <= 0) {
265 BIO_free_all(bio);
266 return false;
267 }
268 signature.resize(actualLen);
269 std::copy(tempSignature.begin(), tempSignature.begin() + actualLen, signature.begin());
270 BIO_free_all(bio);
271 } catch (...) {
272 if (bio) BIO_free_all(bio);
273 return false;
274 }
275 }
276
277 FILE *pubKeyFile = fopen(publicKeyPath.c_str(), "r");
278 if (!pubKeyFile) return false;
279
280 EVP_PKEY *pubKey = PEM_read_PUBKEY(pubKeyFile, nullptr, nullptr, nullptr);
281 fclose(pubKeyFile);
282 if (!pubKey) return false;
283
284 EVP_MD_CTX *ctx = EVP_MD_CTX_new();
285 if (!ctx) {
286 EVP_PKEY_free(pubKey);
287 return false;
288 }
289
290 if (EVP_VerifyInit_ex(ctx, EVP_sha256(), nullptr) != 1 ||
291 EVP_VerifyUpdate(ctx, hash.c_str(), hash.length()) != 1) {
292 EVP_MD_CTX_free(ctx);
293 EVP_PKEY_free(pubKey);
294 return false;
295 }
296
297 int result = EVP_VerifyFinal(ctx, signature.data(), signature.size(), pubKey);
298 EVP_MD_CTX_free(ctx);
299 EVP_PKEY_free(pubKey);
300
301 return (result == 1);
302}
